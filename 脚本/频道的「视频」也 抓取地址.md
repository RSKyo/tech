1、打开 YouTube 频道的【视频】页面；
2、按下快捷键打开控制台：`Command + Option + J`
3、复制以下脚本到控制台并运行；

```js
// 适配频道“视频”页：自动滚动到底 + 仅抓 /watch 视频 + 按 v 去重 + 可选排除短时长 + 复制结果
(async () => {
  const sleep = ms => new Promise(r => setTimeout(r, ms));

  // === 可配置（按需修改）===
  const EXCLUDE_SHORTS = true;     // 强制排除 /shorts/（视频页通常不该有，但保险起见）
  const MIN_SECONDS   = 0;         // 过滤阈值：只保留 >= 指定秒数的视频（0 表示不过滤）
                                   // 若想更接近“ >1分钟 ”的概念，可设为 60
  const WAIT_MS       = 1200;      // 每轮滚动等待时间
  const IDLE_ROUNDS   = 4;         // 连续几轮无新增就停止

  // 简单浮窗（很多环境会拦 alert）
  const toast = (msg) => {
    let t = document.getElementById('__yt_videos_toast__');
    if (!t) {
      t = document.createElement('div');
      t.id = '__yt_videos_toast__';
      Object.assign(t.style, {
        position: 'fixed', right: '16px', bottom: '16px', zIndex: 2147483647,
        maxWidth: '60vw', padding: '10px 14px', borderRadius: '8px',
        background: 'rgba(0,0,0,.85)', color: '#fff', fontSize: '14px', lineHeight: '1.4'
      });
      document.body.appendChild(t);
    }
    t.textContent = msg;
    t.style.display = 'block';
    clearTimeout(t.__hidetimer);
    t.__hidetimer = setTimeout(() => (t.style.display = 'none'), 3000);
  };

  // 解析时长文本 "12:34" / "1:02:03" -> 秒；Shorts 常无时长或 <60s
  const parseDuration = (txt) => {
    if (!txt) return null;
    const s = txt.trim().replace(/\s/g, '');
    if (!/^\d{1,2}:\d{2}(:\d{2})?$/.test(s)) return null;
    const parts = s.split(':').map(n => parseInt(n, 10));
    if (parts.length === 2) return parts[0]*60 + parts[1];
    if (parts.length === 3) return parts[0]*3600 + parts[1]*60 + parts[2];
    return null;
  };

  // 从“视频”网格里抽取：ytd-rich-grid-media 是关键渲染单元
  const collect = () => {
    const cards = document.querySelectorAll('ytd-rich-grid-media');
    const results = [];

    cards.forEach(card => {
      // 链接可能在 #thumbnail 或 #video-title-link
      const a =
        card.querySelector('a#video-title-link[href]') ||
        card.querySelector('a#thumbnail[href]');
      if (!a) return;

      const hrefRaw = a.href || a.getAttribute('href');
      if (!hrefRaw) return;

      const url = new URL(hrefRaw, 'https://www.youtube.com');

      // 保险：排除 shorts
      if (EXCLUDE_SHORTS && url.pathname.startsWith('/shorts/')) return;

      // 只收 /watch?v=xxx
      if (url.pathname !== '/watch') return;
      const v = url.searchParams.get('v');
      if (!v) return;

      // 读出可见的“时长角标”，过滤超短或无角标（直播/短视频常无角标）
      let seconds = null;
      const badge = card.querySelector('ytd-thumbnail-overlay-time-status-renderer span');
      if (badge) {
        seconds = parseDuration(badge.textContent);
      }

      // 过滤逻辑（如果你想尽可能对齐“常规视频”，设 MIN_SECONDS=60）
      if (MIN_SECONDS > 0) {
        if (seconds == null || seconds < MIN_SECONDS) return;
      }

      results.push({
        id: v,
        url: `https://www.youtube.com/watch?v=${v}`,
      });
    });

    return results;
  };

  // 自动滚到底；同时尝试点击“继续浏览更多”之类的续载按钮（旧布局）
  const tryClickContinuation = () => {
    const btn =
      document.querySelector('ytd-continuation-item-renderer a[href], ytd-button-renderer[is="paper-button"] a[href]');
    if (btn) { btn.click(); return true; }
    return false;
  };

  // 主循环：滚动 → 等待 → 收集 → 去重
  toast('⬇️ 正在加载与收集频道视频…');
  const map = new Map(); // key: video id, val: url
  let idle = 0, last = 0;

  for (let i = 0; i < 9999; i++) {
    window.scrollTo(0, document.documentElement.scrollHeight);
    tryClickContinuation();
    await sleep(WAIT_MS);

    const items = collect();
    items.forEach(({id, url}) => map.set(id, url));

    const size = map.size;
    if (size === last) idle++; else { idle = 0; last = size; }
    toast(`已发现 ${size} 个视频…`);
    if (idle >= IDLE_ROUNDS) break;
  }

  const list = Array.from(map.values());
  const text = list.join('\n');

  // 复制到剪贴板（多重兜底）
  let copied = false;
  try { await navigator.clipboard.writeText(text); copied = true; } catch (_) {}
  if (!copied && typeof copy === 'function') { try { copy(text); copied = true; } catch (_) {} }

  console.clear();
  console.log(text);
  toast(`✅ 完成：共 ${list.length} 个链接${copied ? '（已复制）' : '。复制失败，请从控制台复制'}`);
})();
```
